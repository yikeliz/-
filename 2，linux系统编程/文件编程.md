#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

struct Test
{
	int len;
	char name;
};
int main(void)
{

	int fd;
	struct Test  data1[2] = {{10,'a'},{11,'b'}};
	struct Test  data2[2];

	fd = open("./file1",O_RDWR|O_CREAT,0600);
		
	write(fd,&data1,sizeof(struct Test)*2);

	read(fd,&data2,sizeof(struct Test)*2);
